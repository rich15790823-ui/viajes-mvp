<form id="form">

  <label>

    Origen (IATA)

    <input type="text" id="origin" value="CUN" required>

  </label>


  <label>

    Destino (IATA)

    <input type="text" id="destination" value="MAD" required>

  </label>


  <label>

    Fecha de salida (YYYY-MM-DD)

    <input type="date" id="date" value="2025-09-01" required>

  </label>


  <label>

    <input type="checkbox" id="roundTrip">

    Ida y vuelta

  </label>


  <label id="returnDateLabel" style="display:none;">

    Fecha de regreso (YYYY-MM-DD)

    <input type="date" id="returnDate">

  </label>


  <label>

    Adultos

    <input type="number" id="adults" value="1" min="1">

  </label>


  <label>

    Moneda

    <select id="currency">

      <option value="USD">USD</option>

      <option value="MXN">MXN</option>

      <option value="EUR">EUR</option>

    </select>

  </label>


  <button type="submit">Buscar</button>

</form>

<script>

const form = document.getElementById('form');

const roundTripCheckbox = document.getElementById('roundTrip');

const returnDateLabel = document.getElementById('returnDateLabel');


roundTripCheckbox.addEventListener('change', () => {

  returnDateLabel.style.display = roundTripCheckbox.checked ? 'block' : 'none';

});


form.addEventListener('submit', async (e) => {

  e.preventDefault();


  const origin = document.getElementById('origin').value;

  const destination = document.getElementById('destination').value;

  const date = document.getElementById('date').value;

  const adults = document.getElementById('adults').value;

  const currency = document.getElementById('currency').value;

  const returnDate = roundTripCheckbox.checked ? document.getElementById('returnDate').value : '';


  let url = `/api/vuelos?origin=${origin}&destination=${destination}&date=${date}&adults=${adults}&currency=${currency}`;

  if (returnDate) {

    url += `&returnDate=${returnDate}`;

  }


  const res = await fetch(url);

  const data = await res.json();

  console.log(data); // Luego aquí podrías renderizar en tu tabla

});

</script>
